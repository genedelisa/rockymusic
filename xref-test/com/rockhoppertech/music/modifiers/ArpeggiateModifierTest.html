48      @Test
49      public void testArpeggiateModifier() {
50  
51          Chord chord = ChordFactory.getChordByFullSymbol("Cmaj7");
52          MIDITrack track = chord.createMIDITrack();
53          logger.debug("track:\n{}", track);
54  
55          List<Double> series = new ArrayList<Double>();
56          series.add(.25);
57          series.add(.5);
58          series.add(.75);
59          series.add(1d);
60          double startBeat = 1d;
61          double duration = 4d;
62          ArpeggiateModifier mod = new ArpeggiateModifier(startBeat, duration,
63                  series,
64                  NoteModifier.Operation.ADD);
65          track.map(mod);
66          logger.debug("arp track:\n{}", track);
67  
68          MIDINote note = track.get(0);
69          int expectedPitch = Pitch.C0;
70          double expectedStart = 1d;
71          assertThat("pitch is C0",
72                  note.getMidiNumber(),
73                  equalTo(expectedPitch));
74          assertThat("start beat is correct",
75                  note.getStartBeat(),
76                  equalTo(expectedStart));
77  
78          note = track.get(1);
79          expectedStart = 1.25;
80          assertThat("start beat is correct",
81                  note.getStartBeat(),
82                  equalTo(expectedStart));
83  
84          note = track.get(2);
85          expectedStart = 1.75;
86          assertThat("start beat is correct",
87                  note.getStartBeat(),
88                  equalTo(expectedStart));
89  
90          note = track.get(3);
91          expectedStart = 2.5;
92          assertThat("start beat is correct",
93                  note.getStartBeat(),
94                  equalTo(expectedStart));
95      }

100         MIDITrack track = chord.createMIDITrack();
101         double startBeat = 1d;
102         double duration = 4d;
103         double gap = .25;
104         ArpeggiateModifier mod = new ArpeggiateModifier(startBeat, duration,
105                 gap);
106         logger.debug("track:\n{}", track);
107         track.map(mod);
108         logger.debug("arp track:\n{}", track);
109 
110         MIDINote note = track.get(0);
111         int expectedPitch = Pitch.C0;

180     @Test
181     public void multiply() {
182         Chord chord = ChordFactory.getChordByFullSymbol("Cmaj7");
183         MIDITrack track = chord.createMIDITrack();
184         double startBeat = 1d;
185         double duration = 8d;
186         double gap = 2d;
187         ArpeggiateModifier mod = new ArpeggiateModifier(startBeat, duration,
188                 gap, Operation.MULTIPLY);
189         logger.debug("track:\n{}", track);
190         track.map(mod);
191         logger.debug("arp track:\n{}", track);
192 
193         MIDINote note = track.get(0);
194         int expectedPitch = Pitch.C0;
195         double expectedStart = 1d;
196         double expectedDuration = duration;
197         assertThat("pitch is C0",
198                 note.getMidiNumber(),
199                 equalTo(expectedPitch));
200         assertThat("start beat is correct",
201                 note.getStartBeat(),
202                 equalTo(expectedStart));
203         assertThat("duration is correct",
204                 note.getDuration(),
205                 equalTo(expectedDuration));
206 
207         note = track.get(1);
208         expectedStart *= gap;
209         expectedDuration = duration - expectedStart + 1;
210         assertThat("start beat is correct",
211                 note.getStartBeat(),
212                 equalTo(expectedStart));
213         assertThat("duration is correct",
214                 note.getDuration(),
215                 equalTo(expectedDuration));
216 
217         note = track.get(2);
218         expectedStart *= gap;
219         expectedDuration = duration - expectedStart + 1;
220         assertThat("start beat is correct",

224                 note.getDuration(),
225                 equalTo(expectedDuration));
226 
227         note = track.get(3);

230         assertThat("start beat is correct",
