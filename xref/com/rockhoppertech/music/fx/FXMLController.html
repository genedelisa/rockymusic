195 	@FXML
196 	void playAction(ActionEvent event) {
197 		MIDIStringParser parser = new MIDIStringParser();
198 
199 		try {
200 			MIDITrack track = parser.parseString(this.midiStringText.getText());
201 			track.setName("track-" + System.currentTimeMillis());
202 			this.trackText.setText(track.toString());
203 			track.play();
204 			currentTrack = track;
205 			addTrackButton.setDisable(false);
206 		} catch (MIDIParserException e) {
207 			logger.error("Cannot create track {}", e.getMessage(), e);
208 			RTIDialog.showMessageDialog("Cannot create track: \n" + e.getMessage());
209 		}
210

213 	@FXML
214 	void initialize() {
215 		assert fileMenuOpen != null : "fx:id=\"fileMenuOpen\" was not injected: check your FXML file 'Scene.fxml'.";
216 		assert fileMenuQuit != null : "fx:id=\"fileMenuQuit\" was not injected: check your FXML file 'Scene.fxml'.";
217 		assert midiStringText != null : "fx:id=\"midiStringText\" was not injected: check your FXML file 'Scene.fxml'.";
218 		assert parseButton != null : "fx:id=\"parseButton\" was not injected: check your FXML file 'Scene.fxml'.";
219 		assert playButton != null : "fx:id=\"playButton\" was not injected: check your FXML file 'Scene.fxml'.";
220 		assert trackList != null : "fx:id=\"trackList\" was not injected: check your FXML file 'Scene.fxml'.";
221 		assert trackText != null : "fx:id=\"trackText\" was not injected: check your FXML file 'Scene.fxml'.";
222 
223 		tracks = FXCollections.observableArrayList();
224 		trackList.setItems(tracks);
225 		trackList.setEditable(true);
226 		trackList.autosize();
227 		trackList.getSelectionModel().selectedItemProperty().addListener(
228 				new ChangeListener<MIDITrack>() {
229 					@Override
230 					public void changed(
231 							ObservableValue<? extends MIDITrack> ov,
232 							MIDITrack oldTrack, MIDITrack newTrack) {
233 						midiStringText.setText(newTrack.toMIDIString());
234 						trackText.setText(newTrack.toString());
235 					}
236 				});
237 
238 		// we want the MIDITrack's name displayed in the ListView
239 		trackList
240 				.setCellFactory(new Callback<ListView<MIDITrack>, ListCell<MIDITrack>>() {
241 					@Override
242 					public ListCell<MIDITrack> call(ListView<MIDITrack> listView) {
243 						return new ListCell<MIDITrack>() {
244 							@Override
245 							public void updateItem(MIDITrack item, boolean empty) {
246 								super.updateItem(item, empty);
247 								if (empty) {
248 									setText(null);
249 									return;
250 								}
251 								if (item != null) {
252 									setText(item.getName());
253 								}
254 							}
255 						};
256 					}
257 				});
258 		Platform.runLater(new Runnable() {
259 			@Override
260 			public void run() {
261 				midiStringText.requestFocus();
262 			}
263 		});
264 
265 	}

268 	void writeFile(File file) {
269 		String content = midiStringText.getText();
270 		Path path = file.toPath();
271 		// Path path = FileSystems.getDefault().getPath("","");
272 		// try {
273 		// Files.write( path,
274 		// content.getBytes(),
275 		// StandardOpenOption.CREATE);
276 		// } catch (IOException e) {
277 		// e.printStackTrace();
278 		// }
279 
280 		Charset charset = Charset.forName("US-ASCII");
281 		try (BufferedWriter writer = Files.newBufferedWriter(path, charset)) {
282 			writer.write(content, 0, content.length());
283 		} catch (IOException x) {
284 			logger.error(x.getLocalizedMessage(), x);
285 		}
286 	}
287 
288 	void readFile(File file) {
289 		Path path = Paths.get(file.getAbsolutePath());
290 		List<String> alllLines = null;
291 		try {
292 			alllLines = Files.readAllLines(path, StandardCharsets.UTF_8);
293 		} catch (IOException ex) {
294 			logger.error(ex.getMessage());
295 		}
296 		if (alllLines != null) {
297 			for (String s : alllLines) {
298 				midiStringText.appendText(s);
299 				midiStringText.appendText("\n");
300 			}
301 		}
302 	}
