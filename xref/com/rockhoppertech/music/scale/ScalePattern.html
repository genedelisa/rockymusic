71      /**
72       * <pre>
73       * Scale scale = ScaleFactory.getScaleByName(&quot;Major&quot;);
74       * int[] pattern = new int[] { 0, 2 };
75       * int numOctaves = 1;
76       * double duration = Duration.Q;
77       * double restBetweenPatterns = Duration.Q;
78       * boolean reverse = false;
79       * ScalePattern scalePattern = new ScalePattern(scale, pattern, C5,
80       *         numOctaves, duration, reverse, restBetweenPatterns);
81       * scalePattern.setUpAndDown(true);
82       * MIDITrack nl = scalePattern.createNoteList(1d);
83       * </pre>
84       * 
85       * @param scale
86       * @param pattern
87       * @param startPitch
88       * @param numOctaves
89       * @param duration
90       * @param reverse
91       * @param restBetweenPatterns
92       */
93      public ScalePattern(Scale scale, int[] pattern, int startPitch,
94              int numOctaves, double duration, boolean reverse,
95              double restBetweenPatterns) {
96          this.scale = scale;
97          this.startPitch = startPitch;
98          this.pattern = Arrays.copyOf(pattern, pattern.length);
99          this.numOctaves = numOctaves;
100         this.duration = duration;
101         this.reverse = reverse;
102         this.restBetweenPatterns = restBetweenPatterns;
103         Date now = new Date();
104         SimpleDateFormat df = new SimpleDateFormat("yyyy MMM dd kk:mm:ss:SSS");
105         this.setDisplayName(df.format(now));
106         // this.setDisplayName(DateFormat.getDateTimeInstance(DateFormat.SHORT,DateFormat.LONG).format(now));
107     }

227     /**
228      * @param gap
229      *            the duration between patterns
230      * @return a MIDITrack containing the patterns
231      */
232     public MIDITrack createMIDITrack(double gap) {
233         MIDITrack track = new MIDITrack();
234         if (this.scale == null) {
235             throw new IllegalStateException("Scale is null");
236         }
237         MIDITrack list = ScaleFactory
238                 .getNoteListPattern(this.getScale(),
239                         this.getPattern(),
240                         this.getScale().getLength(),
241                         this.getStartPitch(),
242                         this.getNumOctaves(),
243                         this.getDuration(),
244                         this.isReverse(),
245                         this.getRestBetweenPatterns(),
246                         this.upAndDown);
247 
248         String s = String
249                 .format(
250                         "%s - pattern %s - start %d octs %d dur %f reverse %b restBetween %f updown %b",
251                         this.getScale().getName(),
252                         ArrayUtils.toString(this.getPattern()),
253                         // arrayToString(this.getPattern()),
254                         this.getStartPitch(),
255                         this.getNumOctaves(),
256                         this.getDuration(),
257                         this.isReverse(),
258                         this.getRestBetweenPatterns(),
259                         this.upAndDown);
260         track.setName(s);
261         s = String
262                 .format(
263                         "Generated from %s scale.%nThe pattern applied was %s%nthe start pitch is %d%nnumber of octaves %d%nduration %f%nis reverse %b%ngap %f%nup and down %b%n",
264                         this.getScale().getName(),
265                         ArrayUtils.toString(this.getPattern()),
266                         this.getStartPitch(),
267                         this.getNumOctaves(),
268                         this.getDuration(),
269                         this.isReverse(),
270                         this.getRestBetweenPatterns(),
271                         this.upAndDown);
272         track.setDescription(s);
273 
274         track.append(list,
275                 gap);
276 
277         // if (this.upAndDown) {
278         // if (this.getScale().isDescendingDifferent()) {
279         // MIDITrack reverse = ScaleFactory.createMIDITrack(this
280         // .getStartPitch() + 12, this.getScale()
281         // .getDescendingIntervals(), 1d, this.getDuration(), this
282         // .getNumOctaves(), true);
283         // notelist.append(reverse);
284         // notelist.sequential();
285         // } else {
286         // MIDITrack reverse = notelist.retrograde();
287         // notelist.append(reverse);
288         // notelist.sequential();
289         // }
290         // }
291 
292         return track;
293     }

325     @Override
326     public String toString() {
327         StringBuilder sb = new StringBuilder();
328         sb.append("Pattern Name:").append(this.displayName).append(' ');
329         sb.append("pattern:").append(this.getPatternAsString()).append(' ');
330         sb.append("startPitch:").append(this.startPitch).append(' ');
331         sb.append("duration:").append(this.duration).append(' ');        
332         sb.append("upAndDown:").append(this.upAndDown).append(' ');
333         sb.append("numOctaves:").append(this.numOctaves).append(' ');        
334         sb.append("reverse:").append(this.reverse).append(' ');                
335         sb.append("restBetweenPatterns:").append(this.restBetweenPatterns);                
336         return sb.toString();
337     }
