45  /**
46   * Data follows MIDI file format for key signatures.
47   * 
48   * <pre>
49   *   FF 59 02 sf mi
50   * </pre>
51   * 
52   * sf = -7 for 7 flats, -1 for 1 flat, etc, 0 for key of c, 1 for 1 sharp, etc.
53   * mi = 0 for major, 1 for minor
54   * 
55   * It's serializable in order to drag and drop notelists.
56   * 
57   * @author <a href="mailto:gene@rockhoppertech.com">Gene De Lisa</a>
58   * 
59   */

138      * @param majorMinor 0 for major, 1 for minor
139      * @param sf number of sharps or flats. flats < 0, Sharps > 0
140      */
141     private KeySignature(int majorMinor, int sf) {
142         mm = (byte) majorMinor;
143         this.sf = (byte) sf;
144         displayName = getKeySignatureName(mm, this.sf);
145         KeySignature.names.put(displayName, this);
146         // TODO this needs to be the name as a key, i.e. cs and df
147         root = getKeySignatureRootName(mm, this.sf);
148         root = root.trim();
149 
150         // String s = String.format("root '%s'for %d %d", this.root,
151         // this.mm, this.sf);
152         // logger.debug(s);
153 
154         if (MAJOR == mm) {
155             majorRoots.put(root, this);
156             KeySignature.major.put(sf, this);
157             if (sf > 0) {
158                 KeySignature.majorSharp.put(sf, this);
159             } else {
160                 KeySignature.majorFlat.put(sf, this);
161             }
162         } else {
163             minorRoots.put(root, this);
164             KeySignature.minor.put(sf, this);
165             if (sf > 0) {
166                 KeySignature.minorSharp.put(sf, this);
167             } else {
168                 KeySignature.minorFlat.put(sf, this);
169             }
170         }
171     }

357         MetaMessage message = null;
358         try {
359             message = new MetaMessage();
360             byte[] a = new byte[2];
361             a[0] = sf;
362             a[1] = mm;
363             message.setMessage(0x59, a, a.length);
364         } catch (InvalidMidiDataException e) {
365             System.err.println(e);
366         }

382         boolean sharps = false;
383         boolean major = false;
384 
385         if (sf < 0) {

394             major = true;

398         String[] sharpMinorKeys = { "A", "E", "B", "F#", "C#", "G#", "D#", "A#" };
399         String[] flatMinorKeys = { "A", "D", "G", "C", "F", "Bb", "Eb", "Ab" };
400         int num = Math.abs(sf);
401         String str = "";
402

404             if (sharps) {
405                 str += sharpKeys[num];
406             } else {
407                 str += flatKeys[num];

411         } else {
412
